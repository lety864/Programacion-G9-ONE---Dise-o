/* 
1. Crea una función que calcule el índice de masa corporal (IMC) de una persona 
a partir de su altura en metros y peso en kilogramos, que se recibirán como parámetros.
*/

function IMC(altura,peso){
    return peso/(altura*altura);
}

let imc = IMC(1.65,85);
console.log(imc);

/*
2. Crea una función que calcule el valor del factorial de un número pasado como parámetro.
n*(n-1)! recursividad (llamada de la funcion misma)
*/

function factorial(numero){
    return numero ? numero * factorial(numero - 1) : 1;
}

let numeroFactorial = factorial(7);
console.log(numeroFactorial);

/*
4. Crea una función que muestre en pantalla el área y el perímetro de una sala rectangular, 
utilizando la altura y la anchura que se proporcionarán como parámetros.
*/

function AreaRect(altura, ancho){
    return altura*ancho;
}

function PerimetroRect(altura,ancho){
    return 2*(altura+ancho);
}

function salaRectancgular(altura,ancho){
    return `El area del rectangulo es: ${AreaRect(altura,ancho)} y el Perimetro es ${PerimetroRect(altura,ancho)}`; 
}

let rectangulo = salaRectancgular(12,20);
console.log(rectangulo);
