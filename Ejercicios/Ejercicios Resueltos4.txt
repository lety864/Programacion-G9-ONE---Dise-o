/*
7. Crea una función que muestre en la consola 
el número más grande y el número más pequeño en una lista.
*/

function NumeroGyP(lista){
    let maxNum = lista[0];
    let minNum = lista[0];
    for(let i=0; i<lista.length; i++){
        if(lista[i] > maxNum){
            maxNum = lista[i];
        }

        if(lista[i] < minNum){
            minNum = lista[i];
        }
    }

    return `El numero maximo es: ${maxNum} y el minimo es: ${minNum}`;
}

let valor = NumeroGyP(numeros);
console.log(valor);

/*
9. Crea una función que devuelva la posición en la lista donde se encuentra un elemento 
pasado como parámetro, o -1 si no existe en la lista.
*/

function returnPosition(lista,valor){
    let posisciones = [];
    if(lista.includes(valor)){
        for(let i=0; i < lista.length; i++){
            if(lista[i] == valor){
                posisciones.push(i);
            }
        }
    }else{
        return -1;
    }

    return posisciones;
}

/*
10. Crea una función que reciba dos listas de números del mismo tamaño y 
devuelva una nueva lista con la suma de los elementos uno a uno.
*/

let lista1 = [1,4,6,7];
let lista2 = [2,5,7,9];

function SumaListaUnoUno(lista1, lista2){
    let nuevaLista = [];
    if(lista1.length == lista2.length){
        for(let i=0; i < lista1.length; i++){
            nuevaLista.push(lista1[i]+lista2[i]);
        }
    }else{
        return 'Las listas no son iguales en tamaño';
    }

    return nuevaLista;
}

let sumaListas = SumaListaUnoUno(lista1,lista2);
console.log(sumaListas);
