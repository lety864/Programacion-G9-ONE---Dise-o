6. Al hacer clic en el botón "suma", pide 2 números y muestra el resultado de la suma en una alerta.
*/

function operacionSuma(){
    let valor1 = parseInt(prompt('Ingresa un numero entero cualquiera'));
    console.log(valor1);
    let valor2 = parseInt(prompt('Ingresa otro numero entero cualquiera'));
    console.log(valor2);

    let resultado = valor1 + valor2;
    alert(`El valor de la suma es ${resultado}`);
}
